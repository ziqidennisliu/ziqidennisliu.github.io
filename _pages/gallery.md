<script>
(function () {
  function getGapPx(rowEl) {
    var gallery = rowEl.closest('.justified-gallery') || rowEl.parentElement;
    var gap = parseFloat(getComputedStyle(gallery).getPropertyValue('--gallery-gap'));
    return Number.isFinite(gap) ? gap : 4;
  }

  function getImageSize(img) {
    return {
      width: img.naturalWidth || parseFloat(img.getAttribute('width')) || 1,
      height: img.naturalHeight || parseFloat(img.getAttribute('height')) || 1
    };
  }

  /* Calculate layout for one row and show blurred preview */
  function layoutRow(rowEl) {
    var imgs = rowEl.querySelectorAll('img');
    if (!imgs.length) return;

    var gap = getGapPx(rowEl);
    rowEl.style.gap = gap + 'px';
    var containerW = rowEl.parentElement.clientWidth;
    var n = imgs.length;
    var sumAspect = 0;

    imgs.forEach(function (img) {
      var size = getImageSize(img);
      sumAspect += size.width / size.height;
    });

    var rowH = (containerW - (n - 1) * gap) / sumAspect;

    Array.prototype.slice.call(imgs).forEach(function (img) {
      var size = getImageSize(img);
      var aspect = size.width / size.height;
      var w = Math.floor(rowH * aspect);
      img.style.width  = w + 'px';
      img.style.height = Math.round(rowH) + 'px';
      img.classList.add('jg-laid-out');
    });
  }

  function layoutAll() {
    document.querySelectorAll('.justified-gallery').forEach(function (gallery) {
      var rows = gallery.querySelectorAll('.jg-row');
      rows.forEach(function (row, idx) {
        var gap = getGapPx(row);
        row.style.marginTop = idx === 0 ? '0px' : (gap + 'px');
        layoutRow(row);
        row.dataset.laidOut = '1';
      });
    });
  }

  function loadFullImage(img) {
    var fullSrc = img.dataset.fullSrc;
    if (!fullSrc || img.dataset.fullLoaded) return;

    img.dataset.fullLoaded = '1';

    var fullImage = new Image();
    fullImage.decoding = 'async';
    fullImage.onload = function () {
      img.src = fullSrc;
      img.classList.add('jg-loaded');
    };
    fullImage.onerror = function () {
      img.src = fullSrc;
      img.classList.add('jg-loaded');
    };
    fullImage.src = fullSrc;
  }

  function setupFullImageLoading(imgs) {
    if (!('IntersectionObserver' in window)) {
      imgs.forEach(loadFullImage);
      return;
    }

    var observer = new IntersectionObserver(function (entries) {
      entries.forEach(function (entry) {
        if (entry.isIntersecting) {
          observer.unobserve(entry.target);
          loadFullImage(entry.target);
        }
      });
    }, { rootMargin: '600px 0px' });

    imgs.forEach(function (img) {
      observer.observe(img);
    });
  }

  function onReady() {
    var allImgs = Array.prototype.slice.call(document.querySelectorAll('.justified-gallery img'));
    if (!allImgs.length) return;

    layoutAll();

    allImgs.forEach(function (img) {
      img.addEventListener('error', function () {
        if (img.dataset.fullSrc && img.src !== img.dataset.fullSrc) {
          loadFullImage(img);
        }
      });
    });

    setupFullImageLoading(allImgs);
  }

  var resizeTimer;
  window.addEventListener('resize', function () {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(layoutAll, 150);
  });

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', onReady);
  } else {
    onReady();
  }
})();
</script>
