<script>
(function () {
  function getGapPx(rowEl) {
    var gallery = rowEl.closest('.justified-gallery') || rowEl.parentElement;
    var gap = parseFloat(getComputedStyle(gallery).getPropertyValue('--gallery-gap'));
    return Number.isFinite(gap) ? gap : 4;
  }

  /* Check if all images in a row already have dimensions (from HTTP headers) */
  function rowHasDimensions(rowEl) {
    var imgs = rowEl.querySelectorAll('img');
    for (var i = 0; i < imgs.length; i++) {
      if (!imgs[i].naturalWidth) return false;
    }
    return true;
  }

  /* Calculate layout for one row and show blurred preview */
  function layoutRow(rowEl) {
    var imgs = rowEl.querySelectorAll('img');
    if (!imgs.length) return;

    var gap = getGapPx(rowEl);
    rowEl.style.gap = gap + 'px';
    var containerW = rowEl.parentElement.clientWidth;
    var n = imgs.length;
    var sumAspect = 0;

    imgs.forEach(function (img) {
      var w = img.naturalWidth  || 1;
      var h = img.naturalHeight || 1;
      sumAspect += w / h;
    });

    var rowH = (containerW - (n - 1) * gap) / sumAspect;

    Array.prototype.slice.call(imgs).forEach(function (img) {
      var aspect = (img.naturalWidth || 1) / (img.naturalHeight || 1);
      var w = Math.floor(rowH * aspect);
      img.style.width  = w + 'px';
      img.style.height = Math.round(rowH) + 'px';
      /* Phase 1: show blurred preview immediately */
      img.classList.add('jg-laid-out');
    });
  }

  /* Try to layout each row as soon as its image dimensions are available */
  function processGalleries() {
    document.querySelectorAll('.justified-gallery').forEach(function (gallery) {
      var rows = gallery.querySelectorAll('.jg-row');
      rows.forEach(function (row, idx) {
        var gap = getGapPx(row);
        row.style.marginTop = idx === 0 ? '0px' : (gap + 'px');

        if (row.dataset.laidOut) return;
        if (rowHasDimensions(row)) {
          layoutRow(row);
          row.dataset.laidOut = '1';
        }
      });
    });
  }

  /* Full re-layout (used on resize) */
  function layoutAll() {
    document.querySelectorAll('.justified-gallery').forEach(function (gallery) {
      var rows = gallery.querySelectorAll('.jg-row');
      rows.forEach(function (row, idx) {
        var gap = getGapPx(row);
        row.style.marginTop = idx === 0 ? '0px' : (gap + 'px');
        if (rowHasDimensions(row)) {
          layoutRow(row);
          row.dataset.laidOut = '1';
        }
      });
    });
  }

  function onReady() {
    var allImgs = document.querySelectorAll('.justified-gallery img');
    if (!allImgs.length) return;

    /*
     * Poll every 50ms for image dimensions (available from HTTP headers
     * long before the full image is downloaded). As soon as all images
     * in a row have dimensions, layout that row and show blurred preview.
     */
    var dimTimer = setInterval(function () {
      processGalleries();
      var pending = document.querySelectorAll('.jg-row:not([data-laid-out])');
      if (!pending.length) clearInterval(dimTimer);
    }, 50);

    /* Safety: stop polling after 15s */
    setTimeout(function () { clearInterval(dimTimer); }, 15000);

    /*
     * Phase 2: mark each image as fully loaded individually.
     * CSS transitions blur(20px) â†’ blur(0) for a smooth sharpen effect.
     */
    allImgs.forEach(function (img) {
      function markLoaded() { img.classList.add('jg-loaded'); }
      if (img.complete && img.naturalWidth) {
        markLoaded();
      } else {
        img.addEventListener('load',  markLoaded);
        img.addEventListener('error', markLoaded);
      }
    });
  }

  var resizeTimer;
  window.addEventListener('resize', function () {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(layoutAll, 150);
  });

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', onReady);
  } else {
    onReady();
  }
})();
</script>
